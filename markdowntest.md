# Linux Commands

## 1. Process Management (Agent & Job Control)
- ps -ef
- pgrep -f <process>
- kill <PID>
- kill -9 <PID>
- top

# 2. File & Directory Management
- ls -ltr
- cd /path
- mkdir folder
- rm -rf file
- cp file1 file2
- mv file1 file2

# 3. Permissions & Ownership

- chmod 755 script.sh
- chmod +x script.sh
- chown user:group file
- ls -l

# 4. System Health (CPU, Memory, Disk)
- df -h
- du -sh *
- free -m
- top

# 5. Log Monitoring & Troubleshooting

- cat file.log
- less file.log
- tail -f file.log
- grep ERROR file.log
